        <h1 class="text-3xl font-bold text-center text-gray-900 mb-2">Викторина по Экономике</h1>
        <p class="text-center text-gray-600 mb-8">Тема: Типы экономических систем. 11 класс.</p>

        <div id="quiz-container">
            <!-- Вопросы будут вставлены сюда с помощью JS -->
        </div>

        <div class="mt-8 text-center">
            <button id="submit-btn" class="bg-blue-600 text-white font-bold py-3 px-8 rounded-lg hover:bg-blue-700 transition-colors duration-300 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-opacity-50">
                Проверить ответы
            </button>
        </div>

        <div id="results-container" class="hidden mt-8 text-center p-6 bg-gray-50 rounded-lg">
            <h2 class="text-2xl font-bold mb-2">Ваш результат:</h2>
            <p id="score" class="text-4xl font-bold text-blue-600"></p>
            <p id="feedback" class="mt-4 text-lg"></p>
        </div>
    </div>

    <script>
        const quizContainer = document.getElementById('quiz-container');
        const submitButton = document.getElementById('submit-btn');
        const resultsContainer = document.getElementById('results-container');
        const scoreElement = document.getElementById('score');
        const feedbackElement = document.getElementById('feedback');

        const questions = [
            {
                question: "1. Что из перечисленного НЕ является фактором производства?",
                options: ["Труд", "Капитал", "Деньги", "Земля"],
                answer: "Деньги"
            },
            {
                question: "2. Какая экономическая система характеризуется государственной собственностью на ресурсы и централизованным планированием?",
                options: ["Рыночная", "Традиционная", "Командная (плановая)", "Смешанная"],
                answer: "Командная (плановая)"
            },
            {
                question: "3. В рыночной экономике ответ на вопрос «Что производить?» определяется в первую очередь:",
                options: ["Правительственными указами", "Исторически сложившимися традициями", "Спросом потребителей", "Решениями глав корпораций"],
                answer: "Спросом потребителей"
            },
            {
                question: "4. Основная проблема экономики заключается в том, что:",
                options: [
                    "Ресурсы безграничны, а потребности ограничены",
                    "Ресурсы ограничены, а потребности безграничны",
                    "Люди не умеют делать правильный выбор",
                    "Государство неэффективно распределяет блага"
                ],
                answer: "Ресурсы ограничены, а потребности безграничны"
            },
            {
                question: "5. Экономическая система, в которой обычаи и традиции определяют практику использования ограниченных ресурсов, называется:",
                options: ["Рыночной", "Командной", "Традиционной", "Смешанной"],
                answer: "Традиционной"
            },
            {
                question: "6. Процесс передачи государственной собственности в частные руки называется:",
                options: ["Национализация", "Приватизация", "Инфляция", "Монополизация"],
                answer: "Приватизация"
            },
            {
                question: "7. Для какой экономической системы характерно сочетание рыночных механизмов и государственного регулирования?",
                options: ["Традиционная", "Командная", "Чисто рыночная", "Смешанная"],
                answer: "Смешанная"
            },
            {
                question: "8. Что такое альтернативная стоимость?",
                options: [
                    "Стоимость самого дешевого товара",
                    "Ценность лучшей из отвергнутых альтернатив при экономическом выборе",
                    "Все затраты на производство товара",
                    "Сумма денег, потраченная на покупку"
                ],
                answer: "Ценность лучшей из отвергнутых альтернатив при экономическом выборе"
            },
            {
                question: "9. Кто является главным действующим лицом в рыночной экономике?",
                options: ["Государство", "Потребитель и производитель", "Банки", "Крупные корпорации"],
                answer: "Потребитель и производитель"
            },
            {
                question: "10. Что изучает макроэкономика?",
                options: ["Поведение отдельных потребителей и фирм", "Экономику в целом (инфляция, безработица, ВВП)", "Функционирование отдельных рынков", "Ценообразование на конкретный товар"],
                answer: "Экономику в целом (инфляция, безработица, ВВП)"
            }
        ];

        // Функция для отображения вопросов
        function renderQuiz() {
            let quizHTML = '';
            questions.forEach((q, index) => {
                quizHTML += `<div class="mb-6" id="question-${index}">`;
                quizHTML += `<p class="text-lg font-semibold mb-4">${q.question}</p>`;
                quizHTML += `<div class="space-y-3">`;
                q.options.forEach(option => {
                    quizHTML += `
                        <label class="quiz-option flex items-center p-3 border-2 border-gray-200 rounded-lg cursor-pointer hover:bg-gray-50 transition-colors">
                            <input type="radio" name="question${index}" value="${option}" class="mr-3 h-5 w-5 text-blue-600 focus:ring-blue-500 border-gray-300">
                            <span>${option}</span>
                        </label>
                    `;
                });
                quizHTML += `</div></div>`;
            });
            quizContainer.innerHTML = quizHTML;
        }

        // Функция для проверки ответов
        submitButton.addEventListener('click', () => {
            let score = 0;
            const radioButtons = quizContainer.querySelectorAll('input[type="radio"]');
            radioButtons.forEach(rb => rb.disabled = true); // Блокируем ответы после проверки

            questions.forEach((q, index) => {
                const questionBlock = document.getElementById(`question-${index}`);
                const options = questionBlock.querySelectorAll('.quiz-option');
                const selectedOption = questionBlock.querySelector(`input[name="question${index}"]:checked`);

                options.forEach(optLabel => {
                    const input = optLabel.querySelector('input');
                    if (input.value === q.answer) {
                        optLabel.classList.add('correct'); // Подсвечиваем правильный ответ
                    }
                });

                if (selectedOption) {
                    const parentLabel = selectedOption.parentElement;
                    if (selectedOption.value === q.answer) {
                        score++;
                    } else {
                        parentLabel.classList.add('incorrect'); // Подсвечиваем неверный выбор
                    }
                }
            });

            scoreElement.textContent = `${score} / ${questions.length}`;
            
            let feedbackText = '';
            if (score === questions.length) {
                feedbackText = 'Отлично! Вы прекрасно разбираетесь в теме.';
            } else if (score >= questions.length * 0.7) {
                feedbackText = 'Хороший результат! Вы неплохо знаете материал.';
            } else if (score >= questions.length * 0.5) {
                feedbackText = 'Неплохо, но стоит повторить некоторые моменты.';
            } else {
                feedbackText = 'Стоит уделить больше времени изучению этой темы.';
            }
            feedbackElement.textContent = feedbackText;

            resultsContainer.classList.remove('hidden');
            submitButton.style.display = 'none'; // Скрываем кнопку после проверки
            window.scrollTo({ top: document.body.scrollHeight, behavior: 'smooth' });
        });

        // Инициализация викторины
        renderQuiz();
    </script>
